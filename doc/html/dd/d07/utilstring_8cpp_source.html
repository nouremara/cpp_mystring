160 int string::compare(const char* lhsCharArray, const char* rhsCharArray)
161 {
162 // convert both pointers from `char*` to `unsigned char*`
163 // needed for the difference calculations
164 const unsigned char* p1 = (const unsigned char*)lhsCharArray;
165 const unsigned char* p2 = (const unsigned char*)rhsCharArray;
166
167 // check if characters differ, or end of the first string (a terminating null) is reached
168 while (*p1 && *p1 == *p2) {
169 // proceed to the next pair of characters
170 ++p1, ++p2;
171 }
172
173 // return the ASCII difference
174 return (*p1 > *p2) - (*p2 > *p1);
175 }
