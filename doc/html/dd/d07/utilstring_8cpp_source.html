184 int string::compare(const char* lhsCharArray, const char* rhsCharArray) {
185 // convert both pointers from `char*` to `unsigned char*`
186 // needed for the difference calculations
187 const unsigned char* p1 = (const unsigned char*)lhsCharArray;
188 const unsigned char* p2 = (const unsigned char*)rhsCharArray;
189
190 // check if characters differ, or end of the first string (a terminating null)
191 // is reached
192 while (*p1 && *p1 == *p2) {
193 // proceed to the next pair of characters
194 ++p1, ++p2;
195 }
196
197 // return the ASCII difference
198 return (*p1 > *p2) - (*p2 > *p1);
199 }

400 void deepCopy(char* rawCharTarget,
401 const char* rawCharSource,
402 size_t destStartPosition,
403 size_t srcEndPosition) {
404 // check and adjust for default values
405 destStartPosition = (destStartPosition == -1)
406 ? util::string::rawSize(rawCharTarget)
407 : destStartPosition;
408 srcEndPosition = (srcEndPosition == -1) ? util::string::rawSize(rawCharSource)
409 : srcEndPosition;
410
411 // deep copy rawCharSource into rawCharTarget beginning at startPosition
412 for (size_t j = 0; j < srcEndPosition; ++j, ++destStartPosition) {
413 rawCharTarget[destStartPosition] = rawCharSource[j];
414 }
415
416 // ensure destination string is null terminated
417 rawCharTarget[destStartPosition] = '\0';
418 }
